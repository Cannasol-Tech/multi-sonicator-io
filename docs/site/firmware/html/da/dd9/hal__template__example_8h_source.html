79class HalTemplate : public GpioHal<HalImpl>,
80 public AdcHal<HalImpl>,

102class AtmegaHal {
103public:
104 // GPIO Implementation
105 static bool gpioInit() { return gpio_init() == GPIO_OK; }
106 static bool gpioSetPinMode(uint8_t pin, uint8_t mode) {
107 gpio_mode_t gpio_mode;
108 switch (mode) {
109 case 0: gpio_mode = GPIO_MODE_INPUT; break;
110 case 1: gpio_mode = GPIO_MODE_OUTPUT; break;
111 case 2: gpio_mode = GPIO_MODE_INPUT_PULLUP; break;
112 default: return false;
113 }
114 return gpio_set_pin_mode(pin, gpio_mode) == GPIO_OK;
115 }
116 static bool gpioDigitalWrite(uint8_t pin, uint8_t value) {
117 gpio_state_t state = (value == 0) ? GPIO_LOW : GPIO_HIGH;
118 return gpio_write_pin(pin, state) == GPIO_OK;
119 }
120 static uint8_t gpioDigitalRead(uint8_t pin) {
121 gpio_state_t state;
122 gpio_read_pin(pin, &state);
123 return (state == GPIO_HIGH) ? 1 : 0;
124 }
125 static bool gpioTogglePin(uint8_t pin) {
126 return gpio_toggle_pin(pin) == GPIO_OK;
127 }
128 static bool gpioSonicatorStart(uint8_t id) {
129 return gpio_sonicator_start(id) == GPIO_OK;
130 }

132 return gpio_sonicator_stop(id) == GPIO_OK;

135 return gpio_sonicator_reset(id) == GPIO_OK;
136 }

138 return gpio_sonicator_read_overload(id, state) == GPIO_OK;
