333 // Map ATMega32A pin numbers to Arduino pin numbers
334 static uint8_t arduinoPin(uint8_t atmega_pin) {
335 // PORTB: PB0-PB7 (pins 1-8) -> Arduino digital 8-15
336 if (atmega_pin >= 1 && atmega_pin <= 8) {
337 return atmega_pin + 7; // PB0=8, PB1=9, ..., PB7=15
338 }
339 // PORTC: PC0-PC7 (pins 22-29) -> Arduino analog A0-A7
340 else if (atmega_pin >= 22 && atmega_pin <= 29) {
341 return atmega_pin - 22; // PC0=A0, PC1=A1, ..., PC7=A7
342 }
343 // PORTD: PD0-PD7 (pins 14-21) -> Arduino digital 0-7
344 else if (atmega_pin >= 14 && atmega_pin <= 21) {
345 return atmega_pin - 14; // PD0=0, PD1=1, ..., PD7=7
346 }
347 // PORTA: PA0-PA7 (pins 33-40) -> Arduino analog A8-A15 (if supported)
348 else if (atmega_pin >= 33 && atmega_pin <= 40) {
349 return atmega_pin - 25; // PA0=A8, etc. (assuming extended analog pins)
350 }
351 return 0xFF; // Invalid pin
352 }

354 // Get Arduino analog pin for ADC channel
355 static int getAnalogPin(uint8_t channel) {
356 // Map ADC channels to Arduino analog pins
357 // Assuming standard Arduino Uno mapping
358 if (channel >= 0 && channel <= 7) {
359 return A0 + channel; // A0-A7
360 }
361 return -1;
362 }

364 // Get PWM pin for channel
365 static int getPwmPin(uint8_t channel) {
366 // Arduino Uno PWM pins: 3, 5, 6, 9, 10, 11
367 // For amplitude control, use PD7 which maps to digital pin 7
368 // But digital pin 7 is not PWM capable, so use pin 9 (PB1)
369 if (channel == 0) {
370 return 9; // Digital pin 9 for amplitude PWM
371 }
372 return -1;
373 }
