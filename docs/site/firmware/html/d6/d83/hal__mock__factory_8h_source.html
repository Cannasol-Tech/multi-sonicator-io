126 // Map sonicator ID to start pins based on pin matrix

187 bool readChannel(uint8_t channel, uint16_t* raw_value) override {
188 if (!initialized || !raw_value || channel == error_channel) return false;
189 *raw_value = channel_values[channel];
190 return true;
191 }
192
193 bool startConversion(uint8_t channel) override {
194 return (initialized && channel != error_channel);
195 }
196
197 bool isConversionComplete(void) override {
198 return initialized;
199 }
200
201 float rawToVoltage(uint16_t raw_value) override {
202 return (raw_value * 5.0f) / 1023.0f; // 10-bit ADC, 5V reference
203 }
204
205 bool readSonicatorPower(uint8_t sonicator_id, float* power_watts) override {
206 if (!power_watts || !initialized) return false;
207 uint16_t raw;
208 if (!readChannel(sonicator_id + 3, &raw)) return false; // ADC channels 4-7
209 float voltage = rawToVoltage(raw);
210 *power_watts = voltage / 0.00544f; // 5.44 mV/W scaling from pin matrix
211 return true;
212 }
213
214 bool readFrequency(float* frequency_hz) override {
215 if (!frequency_hz || !initialized) return false;
216 uint16_t raw;
217 if (!readChannel(0, &raw)) return false; // LM2907 frequency output
218 float voltage = rawToVoltage(raw);
219 *frequency_hz = voltage * 4000.0f; // Example scaling
220 return true;
221 }
222
223 // Test helpers
224 void setChannelValue(uint8_t channel, uint16_t value) {
225 channel_values[channel] = value;
226 }
227
228 void setErrorChannel(uint8_t channel) { error_channel = channel; }
229 void clearErrorChannel() { error_channel = 255; }
230
231 bool isInitialized() { return initialized; }
232
233 void reset() {
234 channel_values.clear();
235 initialized = false;
236 error_channel = 255;
237 }
