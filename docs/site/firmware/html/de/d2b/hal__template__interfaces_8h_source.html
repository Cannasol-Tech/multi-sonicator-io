195template<typename HalImpl>
196class HalTemplate : public GpioHal<HalImpl>,
197 public AdcHal<HalImpl>,
198 public PwmHal<HalImpl>,
199 public TimerHal<HalImpl>,
200 public UartHal<HalImpl> {
201public:
202 // Unified initialization
203 bool begin() {
204 bool gpio_ok = GpioHal<HalImpl>::init();
205 bool adc_ok = AdcHal<HalImpl>::init();
206 bool pwm_ok = PwmHal<HalImpl>::init();
207 bool timer_ok = TimerHal<HalImpl>::init();
208 // UART initialization requires baud rate, so skip here
209 return gpio_ok && adc_ok && pwm_ok && timer_ok;
210 }
211
212 // Individual peripheral initialization
213 bool beginGpio() { return GpioHal<HalImpl>::init(); }
214 bool beginAdc() { return AdcHal<HalImpl>::init(); }
215 bool beginPwm() { return PwmHal<HalImpl>::init(); }
216 bool beginTimer() { return TimerHal<HalImpl>::init(); }
217 bool beginUart(uint32_t baud_rate) { return UartHal<HalImpl>::init(baud_rate); }
218};
